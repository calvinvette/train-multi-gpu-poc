export USE_CUDA=1

#replace 12.8 with your cuda version
export CUDA_HOME=/usr/local/cuda-12.8
export USE_SYSTEM_NCCL=0
export USE_SYSTEM_CUDNN=1
export BUILD_CAFFE2=0

#replace 8.9 with the compute capability of your gpu for minimal install. Can also be omitted

export MAX_JOBS=16
